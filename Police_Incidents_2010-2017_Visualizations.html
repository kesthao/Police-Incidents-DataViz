SMALL_SIZE = 14
MEDIUM_SIZE = 18
BIGGER_SIZE = 24

#---------------------------------------------------------------------
# PLOTTING START HERE
#---------------------------------------------------------------------
fig = plt.figure(figsize=(15, 15))
# plt.rc('figure', titlesize=TITLE_SIZE)  # fontsize of the figure title
plt.rc('font', size=SMALL_SIZE)          # controls default text sizes

ax = fig.add_axes([0.1, 0.1, 0.8, 0.8])

# Plot
for i,c in enumerate(top_10_incidents.index.tolist()):
    ax.plot(df_top10_grouped[df_top10_grouped['Description']==c]['Day_of_Week_Num'], 
            df_top10_grouped[df_top10_grouped['Description']==c]['Count'], 'o-', 
            lw=3.0, ms=10.0, label=c, c=color_scheme[i])

#------------------------
# X-Axis
#------------------------
plt.xlabel('Day of Week', fontdict=font)
# ax.set_xticklabels(['Mon', 'Tues', 'Wed', 'Thurs', 'Fri', 'Sat', 'Sun'])
# ax.xaxis.set_ticks_position('bottom')
plt.rc('xtick', labelsize=SMALL_SIZE)    # fontsize of the tick labels

#------------------------
# Y-Axis
#------------------------
plt.ylabel('# of Incidents', fontdict=font) # y-label
ax.get_yaxis().set_major_formatter(
    mpl.ticker.FuncFormatter(lambda x, p: format(int(x), ',')))
# ax.set_yticks()

# ax.yaxis.set_ticks_position('left')

# Axes attributes

plt.rc('axes', labelsize=MEDIUM_SIZE)    # fontsize of the x and y labels

# ax.spines['top'].set_visible(False)
# ax.spines['left'].set_visible(False)
# ax.spines['bottom'].set_visible(False)

# Legend
#------------------------
